// 链表节点结构体
struct MyLink {
    float Data;
    struct MyLink* pNext;  // 指向下一个节点的指针
};

// 创建链表节点
struct MyLink* createNode(float data) {
    struct MyLink* node = (struct MyLink*)malloc(sizeof(struct MyLink));
    if (node != NULL) {
        node->Data = data;
        node->pNext = NULL;
    }
    return node;
}

// 在链表末尾插入节点
void insertNode(struct MyLink** ppHead, float data) {
    struct MyLink* newNode = createNode(data);
    if (newNode == NULL) {
        printf("Memory allocation failed.\n");
        return;
    }

    if (*ppHead == NULL) {
        // 链表为空，新节点为头节点
        *ppHead = newNode;
    }
    else {
        // 找到链表末尾
        struct MyLink* current = *ppHead;
        while (current->pNext != NULL) {
            current = current->pNext;
        }
        // 插入新节点
        current->pNext = newNode;
    }
}

####### 汇编指令小计

CMP             R3, #0x63 ; 'c'
ITE EQ
MOVEQ           R3, #1
MOVNE           R3, #0
UXTB            R3, R3
STRB.W          R3, [R7,#0x210+var_1]
B               loc_104B6


CMP R3, #0x63 ; ‘c’
这条指令比较寄存器R3中的值与十六进制数0x63（对应ASCII字符 ‘c’ 的值）。

ITE EQ
根据上一条CMP指令的比较结果（是否相等），设置条件执行标志位（If-Then-Else）。

MOVEQ R3, #1
如果上一条CMP指令的比较结果为相等（即R3等于0x63），则将寄存器R3的值设置为1。

MOVNE R3, #0
如果上一条CMP指令的比较结果为不相等（即R3不等于0x63），则将寄存器R3的值设置为0。

UXTB R3, R3
将寄存器R3中的值进行零扩展字节（Unsigned Extend Byte），确保R3的最低8位是有效的。

STRB.W R3, [R7,#0x210+var_1]
将寄存器R3中的值以字节（Byte）的形式存储到内存地址[R7 + 0x210 + var_1]处。这里的“.W”表示使用字节写入（Word写入），即一次写入8位。

B loc_104B6
无条件跳转到标签loc_104B6处继续执行后续指令。

综上，这段代码的作用是：
比较寄存器R3中的值是否等于ASCII字符 ‘c’ 的ASCII码值（0x63）。
根据比较的结果，将寄存器R3的值设置为1或者0。
将R3的最低8位存储到内存中指定的位置。
最后无条件跳转到另一个代码段（loc_104B6）。
这种结构常用于条件判断并根据条件执行不同的操作，然后将结果存储到内存中，以便后续使用。

UXTB
语法格式：UXTB{cond} {Rd}, Rm {,rotation}
每个32位整型都有4个字节，该命令主要将4个字节的其中一个字节提取出来，然后转成一个新的32位整型
说明：
cond：是一个可选的条件代码。
Rd：是目标寄存器。
Rm：是保存要扩展的值的寄存器。
rotation：是以下其中之一：
ROR #8   Rm 循环右移8 位
ROR #16  Rm循环右移 16 位
ROR #24  Rm循环右移24 位
AAPCS（32位ARM调用约定）

AAPCS（32位ARM调用约定）的主要特点包括：
寄存器使用：

常用的寄存器用途如下：
R0-R3：用于传递前4个整数参数。
R0：用于返回函数的整数结果。
R4-R11：被称为“可选寄存器”，在函数调用中可以被修改，但需要在返回前恢复原值。
堆栈：

如果参数多于4个或者某些参数是浮点数，则额外的参数通过堆栈传递。
函数调用后，返回地址保存在堆栈上，堆栈指针（SP）在函数调用前后必须正确地调整。
浮点数参数和返回值：

浮点数参数通常使用浮点寄存器 S0-S3 或 D0-D7 传递。
浮点数返回值通常使用 S0 或 D0 返回。
对齐要求：

参数和局部变量通常要求按照特定的对齐规则分配，以保证内存访问效率和正确性。
AAPCS64（64位ARM调用约定）的主要特点包括：
寄存器使用：

常用的寄存器用途如下：
X0-X7：用于传递前8个整数参数。
X0：用于返回函数的整数结果。
X8-X18：被称为“可选寄存器”，在函数调用中可以被修改，但需要在返回前恢复原值。
堆栈：

参数多于8个或者某些参数是浮点数时，额外的参数通过堆栈传递。
函数调用后，返回地址和其他保存的寄存器值保存在堆栈上，堆栈指针（SP）需要在函数调用前后正确地调整。
浮点数参数和返回值：

浮点数参数通常使用浮点寄存器 V0-V7 传递。
浮点数返回值通常使用 V0 返回。
对齐要求：

64位数据类型和内存地址通常需要8字节对齐，而32位数据类型需要4字节对齐。
